create table injury_report
(
injury_report_id		int		not null  primary key,
orig_version_id			int		not null,
athlete_id			int		not null  references athlete (athlete_id) on delete cascade,
sport				varchar (50),
position			varchar (50),
body_part			varchar (50),
body_part_detail		text,
injury_type			varchar (50),
injury_type_detail		text,
date_occurred			date,
date_reported			date,
injury_course			varchar (50),
injury_environment		varchar (50),
injury_environment_other	text,
history				text,
visual_inspection		text,
palpation			text,
circulatory_neurological	text,
range_of_motion			text,
strength			text,
special_tests			text,
functional			text,
impression_evaluation		text,
initial_treatment		text,
doctor_referral			text,
injury_status                   varchar (10),
addtnl_remarks			text,
review_required			varchar (3),
confidential			varchar (3),
form_sign_user_id		varchar (50)	not null,
form_sign_date			timestamp	not null,
form_signature			text,
associated_injuries             varchar (500)
);
